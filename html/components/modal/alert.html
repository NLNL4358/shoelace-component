    </sl-alert>
자동으로 사라지는 Alert
sl-alert에 duration="3000" 속성을 사용하면 열리고 3초후 자동으로 닫힙니다 

    <sl-alert variant="primary" duration="3000" open closable>
        <sl-icon slot="icon" name="info-circle"></sl-icon>
        sl-alert에 duration="3000" 속성을 사용하면 열리고
        3초후 자동으로 닫힙니다
    </sl-alert>
열고 닫을 수 있는 Alert - 열때 JS필요!
Show Alert sl-alert 에 .show() 명령어로 Open가능 

    <sl-button id="alert-open">Show Alert</sl-button>

    <sl-alert variant="primary" duration="3000" closable id="openableAlert">
        <sl-icon slot="icon" name="info-circle"></sl-icon>
        sl-alert 에 .show() 명령어로 Open가능
    </sl-alert>
    
    <script>
        const openButton = document.getElementById("alert-open")
        const openableAlert = document.getElementById("openableAlert")

        openButton.addEventListener("click", () => {
            openableAlert.show()
        })
    </script>
카운트다운이 Alert창에서 확인 할 수 있는 Alert
Show Alert sl-alert에 countdown="rtl" 속성을 추가 

    <sl-button id="alert-open2">Show Alert</sl-button>

    <sl-alert variant="primary" duration="3000" countdown="rtl" closable id="countdownAlert">
        <sl-icon slot="icon" name="info-circle"></sl-icon>
        sl-alert에 countdown="rtl" 속성을 추가
    </sl-alert>

    <script>
        const openButton2 = document.getElementById("alert-open2")
        const countdownAlert = document.getElementById("countdownAlert")

        openButton2.addEventListener("click", () => {
            countdownAlert.show()
        })
    </script>
우측 상단에 Toast 식으로 Alert 띄우기
Show Alert alert를 show()로 여는게아닌 toast() 함수로 열게되면 우측상단에 보입니다. 

    <sl-button id="alert-open3">Show Alert</sl-button>

    <sl-alert variant="primary"  duration="3000" countdown="rtl" id="toastAlert" closable>
        <sl-icon slot="icon" name="info-circle"></sl-icon>
        alert를 show()로 여는게아닌 toast() 함수로 열게되면 우측상단에 보입니다.
    </sl-alert>

    <script>
        const openButton3 = document.getElementById("alert-open3")
        const toastAlert = document.getElementById("toastAlert")

        openButton3.addEventListener("click", () => {
            toastAlert.toast() /* .toast() 로 엽니다! */
        })
    </script>
