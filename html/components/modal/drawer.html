        const defaultDrawerOpen = document.getElementById("default-drawer-open")

        defaultDrawerClose.addEventListener("click", ()=>{

영역 밖 터치 방지⚠️ Drawer -> JS로 구현
Open Drawer

drawer에 'sl-request-close' 이벤트를 확인하여 event의 .detail.source === 'overlay'인 경우 event.preventDefault 합니다.

CLose Drawer 

    <sl-button id="deny-drawer-open">Open Drawer</sl-button>

    <sl-drawer label="Deny Drawer" id="deny-drawer" style="--size: 30vw">
        <div class="drawerInner">
            <p>drawer에 'sl-request-close' 이벤트를 확인하여 event의 .detail.source === 'overlay'인 경우 event.preventDefault 합니다.</p>
        </div>
        <sl-button id="deny-drawer-close" slot="footer">CLose Drawer</sl-button>
    </sl-drawer>

    <script>
        const denyDrawerOpen = document.getElementById("deny-drawer-open")
        const denyDrawerClose = document.getElementById("deny-drawer-close")
        const denyDrawer = document.getElementById("deny-drawer")

        denyDrawer.addEventListener('sl-request-close', (e) => {
            if(e.detail.source === 'overlay'){
                console.log("외부 클릭이기에 not close!")
                e.preventDefault();
            }
        })

        denyDrawerOpen.addEventListener("click", () => {denyDrawer.show()})
        denyDrawerClose.addEventListener("click", () => {denyDrawer.hide()})
    </script>
