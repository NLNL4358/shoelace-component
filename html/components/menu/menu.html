    <sl-menu class="menu-value" style="max-width: 200px;">
        <sl-menu-label>option</sl-menu-label>
        <sl-menu-item value="opt-1">Option 1</sl-menu-item>
        <sl-menu-item value="opt-2">Option 2</sl-menu-item>
        <sl-menu-item value="opt-3">Option 3</sl-menu-item>
        <sl-divider></sl-divider>
        <sl-menu-label>checkbox</sl-menu-label>
        <sl-menu-item type="checkbox" value="opt-4">Checkbox 4</sl-menu-item>
        <sl-menu-item type="checkbox" value="opt-5">Checkbox 5</sl-menu-item>
        <sl-menu-item type="checkbox" value="opt-6">Checkbox 6</sl-menu-item>
    </sl-menu>
    
    <script>
        const menu = document.querySelector('.menu-value');
        const menuValue = document.getElementById('value')
    
        menu.addEventListener('sl-select', event => {
        const item = event.detail.item;
    
        // Log value
        if (item.type === 'checkbox') {
            menuValue.innerText = (`Selected value: ${item.value} (${item.checked ? 'checked' : 'unchecked'})`);
        } else {
            menuValue.innerText = (`Selected value: ${item.value}`);
        }
        });
    </script>
