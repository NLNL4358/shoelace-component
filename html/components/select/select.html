선택 value 가져오기


    <div style="display: flex; gap: 2rem">
        <sl-select
            id="select"
        >
            <sl-option value="option-1">Option 1</sl-option>
            <sl-option value="option-2">Option 2</sl-option>
            <sl-option value="option-3">Option 3</sl-option>
        </sl-select>

        <p>선택 value 가져오기</p>
        <p id="select-value"></p>

        <script>
            const select = document.getElementById('select');
            const selectValue = document.getElementById('select-value');

            /* 특이하게 shoelace에서는 sl-change 라는 이벤트를 감지한다. */
            select.addEventListener('sl-change', (e) => {
                selectValue.innerText = e.target.value;
            });

        </script>
    </div>

        .red-select sl-option::part(base):hover{
            /* hover된 option */
            background-color: var(--sl-color-red-200);
        }
        .red-select sl-option[tabindex="0"]::part(base),
        .red-select sl-option[tabindex="0"]::part(base):hover {

            background-color: var(--sl-color-red-500);
