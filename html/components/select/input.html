Value 가져오기 : 


    <sl-input id="sl-input"></sl-input>

    <p>Value 가져오기 : </p>
    <p id="input-value"></p>

    <script>
        const slInput = document.getElementById("sl-input");
        const inputValue = document.getElementById("input-value");

        /* 특이하게 shoelace에서는 sl-input 라는 이벤트를 감지한다. */
        slInput?.addEventListener("sl-input", (e) => {
            inputValue.innerText = e.target.value
        })
    </script>
