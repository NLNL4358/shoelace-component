            <sl-radio value="3">Option 3</sl-radio>
        </sl-radio-group>

        <p>Value 가져오기 </p>
        <p id="radio-value"></p>
    </div>

    <script>
        const radio = document.getElementById("radio");
        const radioValue = document.getElementById("radio-value");

        radio?.addEventListener("sl-change", (e) => {
            radioValue.innerHTML = e.target.value;
        })
    </script>
