    <script>
        const div = document.querySelector('.tags-removable');
    
        div.addEventListener('sl-remove', event => {
        const tag = event.target;
        tag.style.opacity = '0';
        setTimeout(() => (tag.style.opacity = '1'), 2000);
        });
    </script>
